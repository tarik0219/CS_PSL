def create_data(m_random_low,m_random_high,s,csize,size):
    mean_x1 = {}
    mean_x2 = {}
    m1 = random.randint(m_random_low,m_random_high)
    m2 = random.randint(m_random_low,m_random_high)
    mean_x1[0] = m1
    mean_x2[0] = m2
    total1 = np.random.normal(m1,s,size)
    total2 = np.random.normal(m2,s,size)
    for i in range(1,csize):
        m1 = random.randint(m_random_low,m_random_high)
        m2 = random.randint(m_random_low,m_random_high)
        mean_x1[i] = m1
        mean_x2[i] = m2
        test1 = np.random.normal(m1,s,size)
        test2 = np.random.normal(m2,s,size)
        

        total1 = np.concatenate((total1, test1), axis=0)
        total2 = np.concatenate((total2, test2), axis=0)
    return total1,total2, mean_x1, mean_x2

def train_linear(X_train,y_train):
    regr = linear_model.LinearRegression()
    regr.fit(X_train, y_train)
    return regr

def test_linear(regr,X_test,y_test):
    error = 0
    for i,test in  enumerate(X_test):
        predict = regr.predict([test])[0][0]
        if predict < .5:
            predict_score = 0.0
        else:
            predict_score = 1.0
        real_score = y_test[i][0]

        if real_score != predict_score:
            error += 1
    return (error/y_test.shape[0])

def train_quadratic(X_train,y_train):
    poly_reg = PolynomialFeatures(degree = 2) 
    X_poly = poly_reg.fit_transform(X_train)
    poly_reg.fit(X_poly, y_train)
    lin_reg_2 = linear_model.LinearRegression()
    lin_reg_2.fit(X_poly, y_train)
    return lin_reg_2, poly_reg

def test_quadratic(lin_reg_2,X_test,y_test, poly_reg):
    error = 0
    for i,test in  enumerate(X_test):
        predict = lin_reg_2.predict(poly_reg.fit_transform([test]))[0][0]#regr.predict([test])[0][0]
        if predict < .5:
            predict_score = 0.0
        else:
            predict_score = 1.0
        real_score = y_test[i][0]

        if real_score != predict_score:
            error += 1
    return (error/y_test.shape[0])

def test_bayes_error(X_test,y_test, mean_1_x1, mean_1_x2,mean_0_x1,mean_0_x2):
    total_error = 0
    s = 10
    for i,X_value in enumerate(X_test):
        y_value = y_test[i]

        num = 0
        den = 0
        for l in range(0,10):
            m_1 = np.zeros(shape = (1,2))
            m_1[0,0] = mean_1_x1[l]
            m_1[0,1] = mean_1_x2[l]

            m_2 = np.zeros(shape = (1,2))
            m_2[0,0] = mean_0_x1[l]
            m_2[0,1] = mean_0_x2[l]

            num += np.exp((-1/(2*s**2))  * np.linalg.norm(m_1 - X_value) ) 
            den += np.exp((-1/(2*s**2))  * np.linalg.norm(m_2 - X_value) ) 

        #print(num/den, y_value[0])
        if num > den:
            predict = 1.0
        else:
            predict = 0.0

        if y_value[0] != predict:
            total_error += 1

    return total_error/y_test.shape[0]

def train_knn(X_train,y_train,k):
    neigh = KNeighborsClassifier(n_neighbors= k)

    return neigh
def test_knn(neigh,X_test,y_test):
    score = neigh.score(X_test, y_test, sample_weight=None)
    return 1 - score

def simulate(x,k):
    lin_train = []
    lin_test = []
    quad_train = []
    quad_test = []
    knn_train = []
    knn_test = []
    bayes_train = []
    bayes_test = []
    for i in range(0,x):
        #create data
        x1, x2, mean_0_x1, mean_0_x2 = create_data(20,100,8,10,510)
        y1, y2, mean_1_x1, mean_1_x2 = create_data(0,80,8,10,510)
        data_X = np.zeros((10200,2))
        data_y = np.zeros((10200,1))
        data_X[0:5100,0] = x1
        data_X[0:5100,1] = x2
        data_X[5100:,0] = y1
        data_X[5100:,1] = y2
        data_y[0:5100] = 0
        data_y[5100:] = 1
        X_train, X_test, y_train, y_test = train_test_split(data_X, data_y, test_size=(200/10200), random_state=42)
        
        #train each 
        regr = train_linear(X_train,y_train)
        regr_q, poly_reg = train_quadratic(X_train,y_train)
        neigh = train_knn(X_train,y_train,16)
        
        #test test
        lin_test.append(test_linear(regr,X_test,y_test))
        quad_test.append(test_quadratic(regr_q,X_test,y_test, poly_reg))
        knn_test.append(test_knn(neigh,X_test,y_test))
        bayes_test.append(test_bayes_error(X_test,y_test, mean_1_x1, mean_1_x2,mean_0_x1,mean_0_x2))
        
        #test train
        lin_train.append(test_linear(regr,X_train,y_train))
        quad_train.append(test_quadratic(regr_q,X_train,y_train, poly_reg))
        knn_train.append(test_knn(neigh,X_train,y_train))
        bayes_train.append(test_bayes_error(X_train,y_train, mean_1_x1, mean_1_x2,mean_0_x1,mean_0_x2))
        
        print(i)
    
    
    return lin_train,lin_test,quad_train,quad_test,knn_train,knn_test,bayes_train,bayes_test
